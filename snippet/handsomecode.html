      },

  {
    id: 4,
    name: 4,
    children: [],
  },
];
let name = findNameByIdInTree(arr, 20);
console.log("name", name);

function findNameByIdInTree(data, id) {
  for (let item of data) {
    console.log(item.id);
    if (item.id === id) {
      return item.name;
    }
    // 如果当前对象有子项children且不为空，则递归调用findNameByIdInTree函数进行查找
    if (item.children && item.children.length > 0) {
      const foundName = findNameByIdInTree(item.children, id);
      if (foundName !== "no") {
        return foundName;
      }
    }
  }
  return "no"; // 如果未找到匹配的id，返回默认值"no"
}

# 手写弹窗控制body不滚动

.disable-scroll {
    overflow-x: hidden;
    overflow-y: scroll;
    position: fixed;
}
$('.login-register').click(function (){
    pos = $(window).scrollTop();
    $body.addClass('disable-scroll')
    $body.css({
        top: -pos + 'px'
    })
    $('.wrapper').fadeIn()
})
$('.closeImg').click(function (){
    $body.removeClass('disable-scroll')
    $(window).scrollTop(pos);
    $('.wrapper').fadeOut()
})
